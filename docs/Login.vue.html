         * data

         * @property uPassword {String} 密码
         */
        data:function () {
            return {
                uName:"",
                uPassword:""
            }
        },
        mounted:()=>{
            console.log("测试")
        },
        methods:{
            submit:function () {

                axios.post(
                    "/api/loginCheck",
                    qs.stringify({
                        uName:this.uName,
                        uPassword:this.uPassword
                    })


                ).then((res)=>{
                    let user=res.data
                    if (user===""){
                        alert("登录失败")
                    }
                    else {
                        window.location.assign("/")
