    export default {
        name: "Navigationbar",
        prop: ["index"],
        /**
         * 导航栏加载的时候，添加页面滚动监听
         */
        mounted() {
            window.addEventListener("scroll", this.scroll)
        },
        /**
         * 导航栏组件的方法
         */
        methods: {
            click: function (event) {
                console.log(event)
            },
            /**
             * 滚动条判断
             */
            scroll: function () {
                let scrolltop = document.documentElement.scrollTop
                if (scrolltop >= 311) {

                    this.$el.style.position = "fixed"
                    this.$el.style.top = "0px"


                } else {
                    this.$el.style.position = "relative"
                }


            }
        }

    }
</script>
